248  public static void runScript(WebDriver webDriver, String element, String elementValue) throws Exception
249  {
250  HashMap<String, String> result = null;

252  {

264  if (status)

285  @SuppressWarnings("unchecked")
286  private static HashMap<String, String> runGroovyScript(WebDriver webDriver, String element, String elementValue) throws Exception
287  {
288  logger.info("Executing groovy script");
289  final ImportCustomizer imports = new ImportCustomizer();
290  imports.addStaticStars("java.lang.Math");
291  imports.addStarImports("groovyx.net.http");
292  imports.addStaticStars("groovyx.net.http.ContentType", "groovyx.net.http.Method");
293 
294  final SecureASTCustomizer secure = new SecureASTCustomizer();
295  secure.setClosuresAllowed(true);
296 
297  final CompilerConfiguration config = new CompilerConfiguration();
298  config.addCompilationCustomizers(imports, secure);
299 
300  GroovyScriptEngine gse = new GroovyScriptEngine(element);
301  gse.setConfig(config);
302  Binding binding = new Binding();
303  binding.setVariable("logger", logger);
304  binding.setVariable("inputMap", map);
305  binding.setVariable("driver", webDriver);
306  gse.run(elementValue, binding);
307  HashMap<String, String> result = (HashMap<String, String>) binding.getVariable("result");
308  return result;
309  }

348  public static void makeRequest(WebDriver webDriver, String url, String responseVar) throws Exception
349  {
350  CloseableHttpClient client = null;
351  CloseableHttpResponse response = null;
352  try
353  {
354  logger.info("Making request to " + url);
355  client = HttpClients.createDefault();
356  HttpGet request = new HttpGet(url);
357  Set<Cookie> cookies = webDriver.manage().getCookies();
358  for (Iterator<Cookie> iterator = cookies.iterator(); iterator.hasNext();)
359  {
360  Cookie cookie = iterator.next();
361  request.setHeader(cookie.getName(), cookie.getValue());
362  }
363 
364  response = client.execute(request);
365  String content = EntityUtils.toString(response.getEntity(), "UTF-8");
366  setVariable(responseVar, content);
367  }
368  catch (Exception e)
369  {
370  throw new ScriptException("Error making request: " + e.getMessage());
371  }
372  finally
373  {
374  if (client != null)
375  {
376  try
377  {
378  client.close();
379  }
380  catch (IOException e)
381  {
382  logger.warn("MakeRequest: Error on client close");
383  }
384  }
385 
386  if (response != null)
387  {
388  try
389  {
390  response.close();
391  }
392  catch (IOException e)
393  {
394  logger.warn("MakeRequest: Error on response close");
395  }
396  }
397 
398  }
