109  }
110  else
111  {
112  webDriver = new FirefoxDriver();
113  }
114  }
115  else if ("chrome".equals(browserName))

117  System.setProperty("webdriver.chrome.driver", browserCfg.getDriverPath());

263  if (driver != null)
264  {
265  try
266  {
267  driver.manage().window().maximize();
268  WebDriverUtil.initializeWindowHandles(driver);
269  driver.get(suiteConfig.getSettings().getServerUrl());
270  executeScripts(driver);
271  logger.info("**** Finished executing the script. WebDriver will close in a few moments");
272  Thread.sleep(10000);
273  }
274  catch (Exception e)
275  {
276  logger.error("Error while running script", e);
277  }
278  finally
279  {
280  driver.quit();
281  logger.info("Webdriver for browser: " + browserCfg + " completed");
282  }
283  }

292  private void executeScripts(WebDriver webDriver) throws ScriptException
293  {
294  Logger reportsLogger = LoggerFactory.getLogger("reports");
295  SimpleDateFormat format = new SimpleDateFormat("yyyyMMdd-HHmmss");
296  String timestamp = format.format(new Date());
297  String browserString = this.browserCfg.getName() + "-" + timestamp;
298  MDC.put("browser", browserString);
299  List<Script> scripts = suiteConfig.getScripts();
300 
301  for (Script script : scripts)
302  {
303  script.setReportsLogger(reportsLogger);
304  if (script.isRun())
305  {
306  logger.info("Running script:" + script.getName());
307  script.execute(webDriver);
308  }
309  else
310  {
311  logger.info("Skipping script:" + script.getName() + " as Execute is false");
