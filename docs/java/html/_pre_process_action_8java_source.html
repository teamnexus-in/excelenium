155  logger.info("Executing preprocess action " + this.getFullyQualifiedName() + " attributeName: " + this.attributeName + " attributeValue: " + webElement.getAttribute(attributeName) );
156  switch (aType)
157  {

161  case ADD_ATTRIBUTE:
162  case SET_ATTRIBUTE:
163  break;
164  }
165  JavascriptExecutor jsExecutor = (JavascriptExecutor) webDriver;
166  String script = "arguments[0]." + this.attributeName + "=arguments[1]";
167  jsExecutor.executeScript(script, webElement, this.attributeValue);
168  stopwatch.stop();
169  logger.info("Executed preprocess action " + aType + " " + this.attributeName + " " + webElement.getAttribute(this.attributeName) + " " + this.attributeValue);
170  reportsLogger.info("Preprocess Action:" + this.getFullyQualifiedName() + " completed in " + stopwatch.elapsed(TimeUnit.MILLISECONDS) + " msecs");
171  }
172  catch (Exception e)

174  logger.error("error", e);
